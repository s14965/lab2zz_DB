CREATE TABLE users (
id INTEGER GENERATED BY DEFAULT AS IDENTITY(START WITH 1, INCREMENT BY 1) PRIMARY KEY,
username VARCHAR(45) NULL,
password VARCHAR(45) NULL,
email VARCHAR(45) NULL,
privileges char(3) NULL
);
